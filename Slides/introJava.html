Declaring a class - Example 2


import java.util.*;

public class Car extends Vehicle{
  int numberOfSeats;

  public Car(int wheels, String reg, int miles, int seats) {
    super(wheels, reg, miles);
    numberOfSeats = seats;
  }

}
